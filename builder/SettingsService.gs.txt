/**
 * @file SettingsService.gs
 * @description アプリケーション全体の設定を管理する。
 */
var SettingsService = {
  _SETTINGS_KEY: 'WORKFLOW_BUILDER_SETTINGS',
  _POLLING_TRIGGER_ID_KEY: 'WORKFLOW_APP_POLLING_TRIGGER_ID',


  /**
   * 現在の設定を取得する。設定が存在しない場合はデフォルト値を返す。
   * @returns {object} 設定オブジェクト
   */
  getSettings: function () {
    const userProperties = PropertiesService.getUserProperties();
    const settingsJson = userProperties.getProperty(this._SETTINGS_KEY);

    // デフォルト設定
    const defaultSettings = {
      logRetentionDays: 90,
      isLogArchiveEnabled: true,
      isPollingEnabled: false,
      pollingInterval: 10
    };

    if (settingsJson) {
      return Object.assign({}, defaultSettings, JSON.parse(settingsJson));
    }

    return defaultSettings;
  },


  /**
   * 新しい設定を保存する。
   * @param {object} newSettings - 保存する新しい設定オブジェクト。
   */
  saveSettings: function (newSettings) {
    const userProperties = PropertiesService.getUserProperties();
    const currentSettings = this.getSettings();
    const updatedSettings = Object.assign(currentSettings, newSettings);
    userProperties.setProperty(this._SETTINGS_KEY, JSON.stringify(updatedSettings));
    return updatedSettings;
  },

  /**
 * ポーリングトリガーを設定または削除する
 * @param {boolean} isEnabled - ポーリングを有効にするか
 * @param {number} interval - 実行間隔（分）
 */
  setupPollingTrigger: function (isEnabled, interval) {
    this.deletePollingTrigger();

    if (isEnabled) {
      const trigger = ScriptApp.newTrigger('processReservationQueue')
        .timeBased()
        .everyMinutes(interval)
        .create();
      PropertiesService.getUserProperties().setProperty(this._POLLING_TRIGGER_ID_KEY, trigger.getUniqueId());
    }
  },

  /**
   * ポーリングトリガーを削除する
   */
  deletePollingTrigger: function () {
    const userProperties = PropertiesService.getUserProperties();
    const triggerId = userProperties.getProperty(this._POLLING_TRIGGER_ID_KEY);

    if (triggerId) {
      ScriptApp.getProjectTriggers().forEach(function (trigger) {
        if (trigger.getUniqueId() === triggerId) {
          ScriptApp.deleteTrigger(trigger);
        }
      });
      userProperties.deleteProperty(this._POLLING_TRIGGER_ID_KEY);
    }
  },

  /**
   * ポーリングトリガーの現在の状態を取得する
   * @returns {{isEnabled: boolean, interval: number}}
   */
  getPollingStatus: function () {
    return this.getSettings();
  }
};
