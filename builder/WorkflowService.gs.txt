/**
 * @file WorkflowService.gs
 * @description ワークフローの保存、読み込み、実行を担当する。
 */
var WorkflowService = {
  _APP_DATA_FOLDER_NAME: 'WorkflowBuilder AppData',
  _WORKFLOWS_ROOT_FOLDER_NAME: 'Workflow',

  /**
   * アプリケーションデータ用のルートフォルダを取得または作成する。
   * @private
   */
  _getAppDataFolder: function () {
    const folders = DriveApp.getFoldersByName(this._APP_DATA_FOLDER_NAME);
    if (folders.hasNext()) {
      return folders.next();
    }
    return DriveApp.getRootFolder().createFolder(this._APP_DATA_FOLDER_NAME);
  },


  /**
   * ワークフロー保存用のルートフォルダを取得または作成する。
   * @private
   */
  _getWorkflowsRootFolder: function () {
    const appDataFolder = this._getAppDataFolder();
    const folders = appDataFolder.getFoldersByName(this._WORKFLOWS_ROOT_FOLDER_NAME);
    if (folders.hasNext()) {
      return folders.next();
    }
    return appDataFolder.createFolder(this._WORKFLOWS_ROOT_FOLDER_NAME);
  },


  /**
   * 指定されたワークフロー名の専用フォルダを取得または作成する。
   * @private
   * @param {string} workflowName - ワークフロー名
   * @returns {GoogleAppsScript.Drive.Folder} ワークフロー専用フォルダ
   */
  _getWorkflowFolder: function (workflowName) {
    const workflowsRootFolder = this._getWorkflowsRootFolder();
    const workflowFolders = workflowsRootFolder.getFoldersByName(workflowName);
    if (workflowFolders.hasNext()) {
      return workflowFolders.next();
    }
    return workflowsRootFolder.createFolder(workflowName);
  },


  /**
   * 保存されている全ワークフローのリストを取得する。
   * 新旧両方の保存形式に対応する。
   */
  listWorkflows: function () {
    const workflowsRootFolder = this._getWorkflowsRootFolder();
    const workflowNames = new Set();

    const subFolders = workflowsRootFolder.getFolders();
    while (subFolders.hasNext()) {
      const workflowFolder = subFolders.next();
      const files = workflowFolder.getFilesByName('workflow.json');
      if (files.hasNext()) {
        workflowNames.add(workflowFolder.getName());
      }
    }

    const rootFiles = workflowsRootFolder.getFilesByType(MimeType.PLAIN_TEXT);
    while (rootFiles.hasNext()) {
      let file = rootFiles.next();
      const fileName = file.getName();
      if (fileName.endsWith('.json') && fileName !== 'workflow.json') {
        workflowNames.add(fileName.replace('.json', ''));
      }
    }

    return Array.from(workflowNames).sort();
  },


  /**
   * 指定されたワークフローの定義(JSON)を読み込む。
   * 新旧両方の保存形式に対応する。
   * @param {string} workflowName
   */
  loadWorkflow: function (workflowName) {
    const workflowsRootFolder = this._getWorkflowsRootFolder();

    const workflowFolders = workflowsRootFolder.getFoldersByName(workflowName);
    if (workflowFolders.hasNext()) {
      const workflowFolder = workflowFolders.next();
      const files = workflowFolder.getFilesByName('workflow.json');
      if (files.hasNext()) {
        return JSON.parse(files.next().getBlob().getDataAsString());
      }
    }

    const oldFormatFiles = workflowsRootFolder.getFilesByName(`${workflowName}.json`);
    if (oldFormatFiles.hasNext()) {
      return JSON.parse(oldFormatFiles.next().getBlob().getDataAsString());
    }

    return null;
  },


  /**
   * ワークフローを専用フォルダ内に 'workflow.json' として保存する。
   * 上書き保存の場合、古いバージョンを自動的にアーカイブする。
   * @param {string} workflowName
   * @param {Array<object>} workflowData
   */
  saveWorkflow: function (workflowName, workflowData) {
    if (!workflowName) {
      throw new Error('ワークフロー名が指定されていません。');
    }
    const workflowFolder = this._getWorkflowFolder(workflowName);
    const fileName = 'workflow.json';
    const content = JSON.stringify(workflowData, null, 2);

    const files = workflowFolder.getFilesByName(fileName);

    if (files.hasNext()) {
      const existingFile = files.next();

      let archiveFolder;
      const archiveFolders = workflowFolder.getFoldersByName('archive');
      if (archiveFolders.hasNext()) {
        archiveFolder = archiveFolders.next();
      } else {
        archiveFolder = workflowFolder.createFolder('archive');
      }

      const timestamp = Utilities.formatDate(new Date(), 'JST', 'yyyyMMdd_HHmmss');
      existingFile.makeCopy(`workflow_${timestamp}.json`, archiveFolder);

      existingFile.setContent(content);

    } else {
      workflowFolder.createFile(fileName, content, MimeType.PLAIN_TEXT);
    }
    return `ワークフロー「${workflowName}」を専用フォルダに保存しました。`;
  },


  /**
   * ワークフローを削除する。
   * 新旧両方の保存形式に対応する。
   * @param {string} workflowName
   */
  deleteWorkflow: function (workflowName) {
    const workflowsRootFolder = this._getWorkflowsRootFolder();
    let deleted = false;

    const folders = workflowsRootFolder.getFoldersByName(workflowName);
    if (folders.hasNext()) {
      folders.next().setTrashed(true);
      deleted = true;
    }

    const oldFormatFiles = workflowsRootFolder.getFilesByName(`${workflowName}.json`);
    if (oldFormatFiles.hasNext()) {
      oldFormatFiles.next().setTrashed(true);
      deleted = true;
    }

    if (deleted) {
      return `ワークフロー「${workflowName}」を削除しました。`;
    } else {
      throw new Error(`ワークフロー「${workflowName}」は見つかりませんでした。`);
    }
  },

  runSingleModule: function (moduleData) {
    let resultLog = `[個別実行] モジュール: "${moduleData.name}"\n`;
    const context = {
      variables: {},
      executionResults: {}
    };
    try {
      if (moduleData.id === 'control_set_variable') {
        if (moduleData.settings.valueSourceType === 'fromModuleResult' || moduleData.settings.valueSourceType === 'fromCell') {
          throw new Error("個別実行では、他のモジュールやセルから値を取得して変数に設定することはできません。");
        }
      }
      resultLog += this._executeSingleModule(moduleData, context, null).log;
    } catch (e) {
      resultLog += `  -> [エラー] ${e.message}\n`;
    }
    return resultLog;
  },

  runWorkflow: function (payload, executionType = '手動') {
    const startTime = new Date();
    const workflowName = payload.name || '名称未設定のワークフロー';
    const workflowData = payload.modules;

    let logContext = LogService.startLog(workflowName, executionType);
    let status = '成功';
    let uiLog = 'ワークフローの実行を開始します...\n';

    const context = {
      variables: {},
      executionResults: {}
    };

    try {
      uiLog += this._runWorkflowRecursive(workflowData, context, logContext.runId, workflowName);
      uiLog += 'ワークフローの実行が完了しました。';
    } catch (e) {
      status = '失敗';
      uiLog += `[致命的なエラー] ${e.message} \nワークフローの実行を中断しました。`;
      LogService.writeDetailLog(logContext.runId, workflowName, 'Workflow Engine', 'engine', '失敗', e.message);
      Logger.log(e.stack);
    } finally {
      const endTime = new Date();
      const executionTime = (endTime.getTime() - startTime.getTime()) / 1000;
      LogService.endLog(logContext.runId, logContext.summaryRow, status, executionTime);
    }
    return uiLog;
  },

  _runWorkflowRecursive: function (modulesData, context, runId, workflowName) {
    let log = '';
    if (!modulesData) return log;

    for (const module of modulesData) {
      if (module.settings && module.settings.isEnabled === false) {
        const skipMessage = `モジュール: "${module.name}" は無効です。`;
        log += `[スキップ] ${skipMessage}\n`;
        LogService.writeDetailLog(runId, workflowName, module.name, module.instanceId, 'スキップ', skipMessage);
        continue;
      }

      log += `[実行中] モジュール: "${module.name}"\n`;
      let moduleStatus = '成功';
      let resultMessage = '';

      try {
        if (module.type === 'container') {
          if (module.id === 'control_if_else') {
            const result = this._executeModuleLogic(module.id, module.settings, context, runId);
            resultMessage = result.log.trim();
            log += result.log;
            if (result.isTrue) {
              log += this._runWorkflowRecursive(module.settings.ifModules, context, runId, workflowName);
            } else {
              log += this._runWorkflowRecursive(module.settings.elseModules, context, runId, workflowName);
            }
          } else if (module.id === 'control_for_each') {
            const result = this._executeModuleLogic(module.id, module.settings, context, runId);
            const list = result.list;
            resultMessage = `対象リスト(${list.length}件)の処理を開始`;
            log += result.log;

            const originalItem = context.variables.item;
            const originalRowValues = context.variables.row_values;
            const originalIndex = context.variables.index;
            list.forEach((item, index) => {
              log += `  -> ループ ${index + 1} / ${list.length} を開始します。\n`;
              context.variables.item = item[0];
              context.variables.row_values = item;
              context.variables.index = index;
              log += this._runWorkflowRecursive(module.settings.loopModules, context, runId, workflowName);
            });
            context.variables.item = originalItem;
            context.variables.row_values = originalRowValues;
            context.variables.index = originalIndex;
          } else if (module.id === 'control_try_catch') {
            const originalError = context.variables.error_message;
            resultMessage = 'エラー監視ブロックを開始';
            try {
              log += this._runWorkflowRecursive(module.settings.tryModules, context, runId, workflowName);
            } catch (e) {
              log += `  -> [エラー検知] ${e.message}\n`;
              log += `[ジャンプ] エラー処理のため "Catch" ブロックに移動します。\n`;
              context.variables.error_message = e.message;
              LogService.writeDetailLog(runId, workflowName, 'Try-Catch', module.instanceId, 'エラー検知', e.message);
              log += this._runWorkflowRecursive(module.settings.catchModules, context, runId, workflowName);
            } finally {
              context.variables.error_message = originalError;
            }
          } else if (module.id === 'control_group') {
            resultMessage = this._executeModuleLogic(module.id, module.settings, context, runId);
            log += `  -> ${resultMessage}\n`;
            log += this._runWorkflowRecursive(module.settings.groupModules, context, runId, workflowName);
            log += `  -> グループ「${module.settings.groupName || '名称未設定'}」を終了します。\n`;
          }
        } else {
          const result = this._executeSingleModule(module, context, runId);
          log += result.log;
          resultMessage = result.log.replace('  -> 成功: ', '').trim();
          if (result.stop) {
            break;
          }
        }
      } catch (e) {
        moduleStatus = '失敗';
        resultMessage = e.message;
        log += `  -> [エラー] ${e.message}\n`;
      }

      LogService.writeDetailLog(runId, workflowName, module.name, module.instanceId, moduleStatus, resultMessage);
    }
    return log;
  },

  _executeSingleModule: function (module, context, runId) {
    let log = '';
    const resolvedSettings = this._resolveVariablesInSettings(module.settings, context.variables);
    const result = this._executeModuleLogic(module.id, resolvedSettings, context, runId);
    if (result !== null && result !== undefined) {
      context.executionResults[module.instanceId] = result;
    }
    log += `  -> 成功: ${result || '完了'}\n`;
    if (module.id === 'control_noop') {
      return { log, stop: true };
    }
    return { log };
  },

  _resolveVariablesInSettings: function (settings, variables) {
    const resolvedSettings = {};
    for (const key in settings) {
      let value = settings[key];
      if (typeof value === 'string') {
        const placeholders = value.match(/{{(.*?)}}/g);
        if (placeholders) {
          placeholders.forEach(placeholder => {
            const varName = placeholder.replace(/{{|}}/g, '');
            if (variables.hasOwnProperty(varName)) {
              value = value.replace(placeholder, variables[varName]);
            } else {
              value = value.replace(placeholder, '');
            }
          });
        }
      }
      resolvedSettings[key] = value;
    }
    return resolvedSettings;
  },

  _executeModuleLogic: function (moduleId, settings, context, runId) {
    const variables = context.variables;
    switch (moduleId) {
      case 'control_set_variable': return module_control_set_variable(settings, context);
      case 'control_run_workflow': return module_control_run_workflow(settings);
      case 'control_run_workflow_once': return module_control_run_workflow_once(settings, context);
      case 'reservation_add_job': return module_reservation_add_job(settings);
      case 'control_if_else': return module_control_if_else(settings, variables);
      case 'control_for_each': return module_control_for_each(settings, context);
      case 'control_try_catch': return 'エラー監視ブロックを実行します。';
      case 'control_group': return `グループ「${settings.groupName || '名称未設定'}」の処理を開始します。`;
      case 'control_noop': return 'この分岐の処理をここで終了します。';
      case 'date_manipulation': return module_date_manipulation(settings);
      case 'log_message': return module_log_message(settings);
      case 'sheets_create_new': return module_sheets_create_new(settings);
      case 'sheets_to_pdf': return module_sheets_to_pdf(settings);
      case 'sheets_to_excel': return module_sheets_to_excel(settings);
      case 'sheets_copy_values': module_sheets_copy_values(settings); break;
      case 'sheets_set_value': module_sheets_set_value(settings); break;
      case 'sheets_insert_image': module_sheets_insert_image(settings); break;
      case 'sheets_clear_values': module_sheets_clear_values(settings); break;
      case 'sheets_import_csv': module_sheets_import_csv(settings); break;
      case 'sheets_insert_rows_before': module_sheets_insert_rows_before(settings); break;
      case 'sheets_insert_rows_after': module_sheets_insert_rows_after(settings); break;
      case 'sheets_delete_rows': module_sheets_delete_rows(settings); break;
      case 'sheets_insert_cols_before': module_sheets_insert_cols_before(settings); break;
      case 'sheets_insert_cols_after': module_sheets_insert_cols_after(settings); break;
      case 'sheets_delete_cols': module_sheets_delete_cols(settings); break;
      case 'sheets_hide_rows_by_value': module_sheets_hide_rows_by_value(settings); break;
      case 'sheets_unhide_rows': module_sheets_unhide_rows(settings); break;
      case 'sheets_get_monthly_data': module_sheets_get_monthly_data(settings); break;
      case 'drive_create_folder': return module_drive_create_folder(settings);
      case 'drive_copy_file': return module_drive_copy_file(settings);
      case 'drive_delete_folder_files': module_drive_delete_folder_files(settings); break;
      case 'drive_delete_files_by_type': module_drive_delete_files_by_type(settings); break;
      case 'drive_list_files_to_sheet': module_drive_list_files_to_sheet(settings); break;
      case 'drive_convert_excel_to_sheets': return module_drive_convert_excel_to_sheets(settings);
      case 'gmail_fetch_attachments': module_gmail_fetch_attachments(settings); break;
      case 'gmail_create_draft': module_gmail_create_draft(settings); break;
      case 'gmail_send_email': module_gmail_send_email(settings); break;
      case 'slides_refresh_charts': module_slides_refresh_charts(settings); break;
      case 'calendar_get_holidays_to_sheet': module_calendar_get_holidays_to_sheet(settings); break;
      case 'chat_post': return module_chat_post(settings);
      case 'utils_extract_id_from_url': return module_utils_extract_id_from_url(settings);
      case 'sheets_append_block': return module_sheets_append_block(settings, context);
      case 'custom_run_script':
        return module_custom_run_script(settings, context);
      default:
        throw new Error(`ID '${moduleId}' に対応する処理関数は定義されていません。`);
    }
    return null;
  },

  _getQueueFolder: function () {
    const appFolder = this._getAppDataFolder();
    const queueFolders = appFolder.getFoldersByName('Queue');
    if (queueFolders.hasNext()) {
      return queueFolders.next();
    }
    return appFolder.createFolder('Queue');
  },
};
