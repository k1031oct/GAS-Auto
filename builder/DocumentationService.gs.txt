/**
 * @file DocumentationService.gs
 * @description ワークフローの構成を解析し、仕様書を生成する。
 */
var DocumentationService = {


  /**
   * メインの処理関数。ワークフローデータを受け取り、ドキュメントを生成する。
   * @param {object} payload - {name: string, modules: Array<object>}
   * @returns {string} 生成されたGoogleドキュメントのURL
   */
  generate: function (payload) {
    const workflowName = payload.name || '名称未設定のワークフロー';
    const modules = payload.modules || [];

    const workflowFolder = WorkflowService._getWorkflowFolder(workflowName);
    const docName = `【仕様書】${workflowName}`;

    const existingFiles = workflowFolder.getFilesByName(docName);
    if (existingFiles.hasNext()) {
      existingFiles.next().setTrashed(true);
    }

    const doc = DocumentApp.create(docName);
    const docFile = DriveApp.getFileById(doc.getId());

    workflowFolder.addFile(docFile);
    DriveApp.getRootFolder().removeFile(docFile);

    const body = doc.getBody();
    this._setDocumentStyles(body);

    body.appendParagraph(`ワークフロー仕様書: ${workflowName}`).setHeading(DocumentApp.ParagraphHeading.TITLE);
    body.appendParagraph(`生成日時: ${new Date().toLocaleString('ja-JP')}`).setHeading(DocumentApp.ParagraphHeading.SUBTITLE);
    body.appendHorizontalRule();

    body.appendParagraph('処理ステップ').setHeading(DocumentApp.ParagraphHeading.HEADING1);

    this._processModules(body, modules, 0);

    doc.saveAndClose();
    return doc.getUrl();
  },


  /**
   * モジュールの配列を再帰的に処理し、ドキュメントに内容を書き出す。
   * @param {GoogleAppsScript.Document.Body} body - 書き込み対象のドキュメントBody
   * @param {Array<object>} modules - 処理対象のモジュール配列
   * @param {number} indentLevel - インデントのレベル
   */
  _processModules: function (body, modules, indentLevel) {
    modules.forEach((module, index) => {
      const description = this._generateModuleDescription(module);

      const listItem = body.appendListItem(`${description.title}`);
      listItem.setGlyphType(DocumentApp.GlyphType.DECIMAL);
      listItem.setIndentStart(18 * indentLevel);

      if (description.summary) {
        body.appendParagraph(description.summary).setIndentStart(18 * (indentLevel + 1));
      }

      if (description.table && description.table.length > 0) {
        const table = body.appendTable(description.table);
        table.setBorderColor('#DDDDDD');
        table.getRow(0).getCell(0).setBackgroundColor('#F3F3F3').setBold(true);
        table.getRow(0).getCell(1).setBackgroundColor('#F3F3F3').setBold(true);
      }

      if (description.code) {
        const cell = body.appendTable([[`${description.code}`]]).getCell(0, 0);
        cell.setBackgroundColor('#F3F3F3');
        const style = {};
        style[DocumentApp.Attribute.FONT_FAMILY] = 'Consolas';
        cell.setAttributes(style);
      }

      if (module.type === 'container') {
        const containerIndent = indentLevel + 1;
        if (module.settings.ifModules) {
          body.appendListItem('IF (真の場合):').setGlyphType(DocumentApp.GlyphType.HOLLOW_BULLET).setIndentStart(18 * containerIndent);
          this._processModules(body, module.settings.ifModules, containerIndent + 1);
        }
        if (module.settings.elseModules) {
          body.appendListItem('ELSE (偽の場合):').setGlyphType(DocumentApp.GlyphType.HOLLOW_BULLET).setIndentStart(18 * containerIndent);
          this._processModules(body, module.settings.elseModules, containerIndent + 1);
        }
        if (module.settings.loopModules) {
          this._processModules(body, module.settings.loopModules, containerIndent);
        }
        if (module.settings.tryModules) {
          body.appendListItem('TRY (エラー監視ブロック):').setGlyphType(DocumentApp.GlyphType.HOLLOW_BULLET).setIndentStart(18 * containerIndent);
          this._processModules(body, module.settings.tryModules, containerIndent + 1);
        }
        if (module.settings.catchModules) {
          body.appendListItem('CATCH (エラー発生時ブロック):').setGlyphType(DocumentApp.GlyphType.HOLLOW_BULLET).setIndentStart(18 * containerIndent);
          this._processModules(body, module.settings.catchModules, containerIndent + 1);
        }
        if (module.settings.groupModules) {
          this._processModules(body, module.settings.groupModules, containerIndent);
        }
      }
    });
  },


  /**
   * 単一のモジュールデータから、仕様書に記載する情報を生成する。
   * @param {object} module - モジュールデータ
   * @returns {{title: string, summary?: string, table?: Array<Array<string>>, code?: string}}
   */
  _generateModuleDescription: function (module) {
    const s = module.settings;
    let title = module.name;
    let summary = '';
    let table = [];
    let code = '';

    const getFileName = (settings) => {
      if (settings.fileNameSourceType === 'cell') {
        return `スプレッドシートのセル「${settings.fileNameSourceCell}」の値`;
      }
      return `「${settings.newFileName || '(未設定)'}」`;
    };

    switch (module.id) {
      // --- ワークフロー制御 ---
      case 'control_set_variable':
        title = `変数を設定`;
        summary = `変数 **${s.variableName || '(未設定)'}** に値を設定します。`;
        if (s.valueSourceType === 'manual') {
          table.push(['値のソース', '手入力']);
          table.push(['設定する値', s.variableValue]);
        } else if (s.valueSourceType === 'fromCell') {
          table.push(['値のソース', 'セルから取得']);
          table.push(['シートURL', s.sourceUrl]);
          table.push(['シート名', s.sourceSheet]);
          table.push(['セル', s.sourceCell]);
        } else {
          table.push(['値のソース', '前のモジュールの実行結果']);
          table.push(['対象モジュールID', s.sourceModuleInstanceId]);
        }
        break;
      case 'control_run_workflow':
        title = `別のワークフローを実行`;
        summary = `ワークフロー **${s.workflowToRun || '(未設定)'}** を非同期で呼び出します。`;
        break;
      case 'control_if_else':
        title = `条件分岐`;
        const operatorMap = {
          equals: 'と等しい', not_equals: 'と等しくない', is_blank: 'が空である', is_not_blank: 'が空でない',
          contains: 'を含む', not_contains: 'を含まない', greater_than: 'より大きい', less_than: 'より小さい'
        };
        summary = `以下の条件を評価し、処理を分岐します。`;
        table.push(['判定する値', s.sourceValue]);
        table.push(['比較演算子', operatorMap[s.operator] || s.operator]);
        table.push(['比較する値', s.comparisonValue]);
        break;
      case 'control_for_each':
        title = `繰り返し (For Each)`;
        summary = '指定されたリストの各項目に対して、後続の処理を繰り返します。';
        if (s.sourceType === 'range') {
          table.push(['リストのソース', 'スプレッドシートの範囲']);
          table.push(['シートURL', s.sourceUrl]);
          table.push(['シート名', s.sourceSheet]);
          table.push(['範囲', s.sourceRange]);
        } else if (s.sourceType === 'text') {
          table.push(['リストのソース', 'カンマ区切りのテキスト']);
          table.push(['テキストリスト', s.listText]);
        } else {
          table.push(['リストのソース', '前のモジュールの実行結果']);
        }
        break;
      case 'control_try_catch': title = `エラー処理 (Try-Catch)`; summary = '後続の処理でエラーが発生した場合に、代替処理を実行します。'; break;
      case 'control_noop': title = `処理を終了`; summary = 'この分岐における以降の処理を中断します。'; break;
      case 'control_group': title = `グループ: ${s.groupName || '名称未設定'}`; summary = '後続のモジュールをグループ化します。'; break;
      case 'log_message':
        title = 'ログを出力';
        summary = `実行ログに以下のメッセージを出力します。`;
        code = s.logMessage;
        break;

      // --- Google Drive & Sheets ---
      case 'drive_copy_file':
        title = 'ファイルをコピー';
        summary = `Google Drive上のファイルをコピーします。`;
        table.push(['項目', '設定値']);
        table.push(['コピー元ファイルID', s.sourceFileId]);
        table.push(['コピー先フォルダID', s.destFolderId]);
        table.push(['新しいファイル名', getFileName(s)]);
        break;
      case 'sheets_to_pdf':
        title = 'PDFとして出力';
        summary = `スプレッドシートの指定されたシートをPDF形式で出力します。`;
        table.push(['項目', '設定値']);
        table.push(['スプレッドシートURL', s.spreadsheetUrl]);
        table.push(['シート名', s.sheetName]);
        table.push(['出力ファイル名', getFileName(s)]);
        table.push(['保存先フォルダID', s.destFolderId]);
        break;
      case 'sheets_copy_values':
        title = '値のコピー＆ペースト';
        summary = '指定されたスプレッドシート間で、セルの値をコピーします。';
        table.push(['項目', '設定値']);
        table.push(['コピー元URL', s.sourceUrl]);
        table.push(['コピー元シート', s.sourceSheet]);
        table.push(['コピー元範囲', s.sourceRange]);
        table.push(['貼付先URL', s.targetUrl]);
        table.push(['貼付先シート', s.targetSheet]);
        table.push(['貼付先範囲(左上)', s.targetRange]);
        break;
      case 'custom_run_script':
        title = 'カスタムスクリプト実行';
        summary = '以下のApps Scriptコードを実行します。';
        code = s.userCode;
        break;


      // --- その他のモジュール (簡易表示) ---
      default:
        Object.keys(s).forEach(key => {
          if (typeof s[key] === 'string' && s[key] && !key.endsWith('Modules')) {
            if (table.length === 0) table.push(['パラメータ', '値']);
            table.push([key, s[key]]);
          }
        });
        break;
    }
    return { title, summary, table, code };
  },


  /**
   * ドキュメントの基本的なスタイルを設定する。
   * @param {GoogleAppsScript.Document.Body} body 
   */
  _setDocumentStyles: function (body) {
    body.setPageWidth(1600);
    body.setPageHeight(2200);

    const style = {};
    style[DocumentApp.Attribute.FONT_FAMILY] = 'Meiryo';
    style[DocumentApp.Attribute.FONT_SIZE] = 10;
    body.setAttributes(style);
  }
};
