/**
 * @file TriggerService.gs
 * @description ワークフローの自動実行トリガーを管理する。
 */
var TriggerService = {

  _HANDLER_FUNCTION_NAME: 'executeWorkflowByTrigger',


  /**
   * 新しい柔軟なトリガーを作成する
   * @param {object} config - フロントエンドから渡されるトリガー設定オブジェクト
   * @returns {object} 作成されたトリガーの情報
   */
  createFlexibleTrigger: function (config) {
    if (!config.workflowName || !config.triggerType) {
      throw new Error('ワークフロー名またはトリガータイプが指定されていません。');
    }

    let triggerBuilder;

    switch (config.triggerType) {
      // --- 時間主導型トリガー ---
      case 'TIME_EVERY_MINUTES':
      case 'TIME_EVERY_HOURS':
      case 'TIME_EVERY_DAYS':
      case 'TIME_DAILY':
      case 'TIME_WEEKLY':
      case 'TIME_MONTHLY':
      case 'TIME_SPECIFIC':
        triggerBuilder = ScriptApp.newTrigger(this._HANDLER_FUNCTION_NAME).timeBased();
        break;

      // --- イベント駆動型トリガー ---
      case 'EVENT_FORM_SUBMIT':
        const formId = _extractIdFromUrl(config.formUrl);
        if (!formId) throw new Error('GoogleフォームのURLからIDを抽出できませんでした。');
        triggerBuilder = ScriptApp.newTrigger(this._HANDLER_FUNCTION_NAME).forForm(formId).onFormSubmit();
        break;

      case 'EVENT_SHEET_EDIT':
        const sheetId = _extractIdFromUrl(config.sheetUrl);
        if (!sheetId) throw new Error('GoogleスプレッドシートのURLからIDを抽出できませんでした。');
        triggerBuilder = ScriptApp.newTrigger(this._HANDLER_FUNCTION_NAME).forSpreadsheet(sheetId).onChange();
        break;

      case 'EVENT_DRIVE_FILE_NEW':
      case 'EVENT_GMAIL_NEW':
        triggerBuilder = ScriptApp.newTrigger(this._HANDLER_FUNCTION_NAME).timeBased().everyHours(1);
        break;

      default:
        throw new Error(`未対応のトリガータイプです: ${config.triggerType}`);
    }

    if (config.triggerType.startsWith('TIME_')) {
      const timeParts = (config.time || '00:00').split(':');
      const hour = parseInt(timeParts[0], 10);
      const minute = parseInt(timeParts[1], 10);

      switch (config.triggerType) {
        case 'TIME_EVERY_MINUTES':
          triggerBuilder.everyMinutes(config.interval);
          break;
        case 'TIME_EVERY_HOURS':
          triggerBuilder.everyHours(config.interval);
          break;
        case 'TIME_EVERY_DAYS':
          triggerBuilder.atHour(hour).nearMinute(15).everyDays(config.interval);
          break;
        case 'TIME_DAILY':
          triggerBuilder.atHour(hour).nearMinute(minute).everyDays(1);
          break;
        case 'TIME_WEEKLY':
          if (!config.days || config.days.length === 0) throw new Error('曜日が指定されていません。');
          config.days.forEach(day => triggerBuilder.onWeekDay(ScriptApp.WeekDay[day]));
          triggerBuilder.atHour(hour).nearMinute(minute);
          break;
        case 'TIME_MONTHLY':
          if (!config.dates || config.dates.length === 0) throw new Error('日付が指定されていません。');
          config.dates.forEach(date => triggerBuilder.onMonthDay(date));
          triggerBuilder.atHour(hour).nearMinute(minute);
          break;
        case 'TIME_SPECIFIC':
          const specificDate = new Date(config.dateTime);
          if (isNaN(specificDate.getTime())) throw new Error('日時の形式が無効です。');
          triggerBuilder.at(specificDate);
          break;
      }
    }

    const trigger = triggerBuilder.create();
    const triggerUid = trigger.getUniqueId();

    const properties = PropertiesService.getUserProperties();
    properties.setProperty(triggerUid, config.workflowName);
    properties.setProperty(`config_${triggerUid}`, JSON.stringify(config));

    Logger.log(`トリガーを作成しました。ID: ${triggerUid}, ワークフロー: ${config.workflowName}, 設定: ${JSON.stringify(config)}`);
    return this._convertTriggerToObject(trigger);
  },


  /**
 * 指定された日時に一度だけ実行されるトリガーを作成する（重複時は自動で上書き）
 * @param {string} workflowName - 実行するワークフローの名前
 * @param {Date} executionDate - 実行する日時
 * @returns {string} 予約完了メッセージ
 */
  createOneTimeTrigger: function (workflowName, executionDate) {
    const properties = PropertiesService.getUserProperties();
    const timezone = Session.getScriptTimeZone();
    const formattedDate = Utilities.formatDate(executionDate, timezone, 'yyyy-MM-dd HH:mm:ss');

    const jobKey = `JOB_${workflowName}_${formattedDate}`;

    const oldTriggerId = properties.getProperty(jobKey);
    if (oldTriggerId) {
      _deleteTriggerById(oldTriggerId);
      Logger.log(`古い予約トリガー (ID: ${oldTriggerId}) を削除し、新しい予約で上書きします。`);
    }

    const trigger = ScriptApp.newTrigger('executeOneTimeWorkflow')
      .timeBased()
      .at(executionDate)
      .create();
    const triggerUid = trigger.getUniqueId();

    const propValue = JSON.stringify({
      workflowName: workflowName,
      jobKey: jobKey
    });
    properties.setProperty(`ONETIME_${triggerUid}`, propValue);
    properties.setProperty(jobKey, triggerUid);

    Logger.log(`1回限りのトリガーを作成しました。ID: ${triggerUid}, ワークフロー: ${workflowName}, 実行日時: ${formattedDate}`);
    return `ワークフロー「${workflowName}」を ${formattedDate} に実行するよう予約しました。`;
  },


  /**
   * 既存のトリガーを更新する（実態は削除＆再作成）
   * @param {object} config - フロントエンドから渡されるトリガー設定オブジェクト（uidを含む）
   * @returns {object} 更新されたトリガーの情報
   */
  updateFlexibleTrigger: function (config) {
    if (!config.uid) {
      throw new Error("更新対象のトリガーID(uid)が指定されていません。");
    }
    this.deleteTrigger(config.uid);
    return this.createFlexibleTrigger(config);
  },

  getTriggers: function () {
    return ScriptApp.getProjectTriggers()
      .filter(trigger => trigger.getHandlerFunction() === this._HANDLER_FUNCTION_NAME)
      .map(trigger => this._convertTriggerToObject(trigger));
  },

  deleteTrigger: function (triggerUid) {
    const triggers = ScriptApp.getProjectTriggers();
    let found = false;
    for (const trigger of triggers) {
      if (trigger.getUniqueId() === triggerUid) {
        ScriptApp.deleteTrigger(trigger);
        found = true;
        break;
      }
    }

    if (!found) {
      Logger.log(`削除対象のトリガーが見つかりませんでした。ID: ${triggerUid}`);
    }

    const properties = PropertiesService.getUserProperties();
    properties.deleteProperty(triggerUid);
    properties.deleteProperty(`config_${triggerUid}`);

    Logger.log(`トリガー関連プロパティを削除しました。ID: ${triggerUid}`);
    return `トリガー(ID: ${triggerUid})を削除しました。`;
  },

  _convertTriggerToObject: function (trigger) {
    const triggerUid = trigger.getUniqueId();
    const properties = PropertiesService.getUserProperties();
    const workflowName = properties.getProperty(triggerUid) || '不明なワークフロー';
    const configStr = properties.getProperty(`config_${triggerUid}`);
    const config = configStr ? JSON.parse(configStr) : {};

    const name = config.triggerName || "名称未設定のトリガー";
    const description = config.triggerDescription || this._generateDescription(config);

    return {
      uid: triggerUid,
      workflowName: workflowName,
      name: name,
      description: description,
      config: config
    };
  },

  _generateDescription: function (config) {
    let desc = '条件不明';
    if (!config || !config.triggerType) {
      Logger.log('説明生成スキップ: configまたはtriggerTypeが未定義です。');
      return desc;
    }

    try {
      switch (config.triggerType) {
        case 'TIME_EVERY_MINUTES':
          desc = `${config.interval || '?'}分ごと`;
          break;
        case 'TIME_EVERY_HOURS':
          desc = `${config.interval || '?'}時間ごと`;
          break;
        case 'TIME_EVERY_DAYS':
          desc = `${config.interval || '?'}日ごと (${config.time || '指定時刻'}頃)`;
          break;
        case 'TIME_DAILY':
          desc = `毎日 ${config.time || '指定時刻'} 頃`;
          break;
        case 'TIME_WEEKLY':
          const days = Array.isArray(config.days) ? `[${config.days.join(',')}]` : '[曜日未設定]';
          desc = `毎週 ${days} の ${config.time || '指定時刻'} 頃`;
          break;
        case 'TIME_MONTHLY':
          const dates = Array.isArray(config.dates) ? `[${config.dates.join(',')}]` : '[日付未設定]';
          desc = `毎月 ${dates} 日の ${config.time || '指定時刻'} 頃`;
          break;
        case 'TIME_SPECIFIC':
          desc = `指定日時: ${config.dateTime || '未設定'}`;
          break;
        case 'EVENT_FORM_SUBMIT':
          desc = `フォーム送信時: ${(config.formUrl || 'URL未設定').substring(0, 40)}...`;
          break;
        case 'EVENT_SHEET_EDIT':
          desc = `シート編集時: ${(config.sheetUrl || 'URL未設定').substring(0, 40)}...`;
          if (config.sheetName) desc += ` [シート: ${config.sheetName}]`;
          if (config.range) desc += ` [範囲: ${config.range}]`;
          break;
        case 'EVENT_DRIVE_FILE_NEW':
          desc = `(1時間毎) ドライブ監視: ${(config.driveFolderUrl || 'URL未設定').substring(0, 40)}...`;
          break;
        case 'EVENT_GMAIL_NEW':
          desc = `(1時間毎) Gmail監視: ${config.gmailQuery || 'クエリ未設定'}`;
          break;
      }
    } catch (e) {
      Logger.log(`説明生成中にエラー発生。config: ${JSON.stringify(config)}, error: ${e.toString()}`);
      desc = '説明生成エラー';
    }
    return desc;
  },

};
