/**
 * @file ReservationService.gs
 * @description 個別修正予約用のスプレッドシートを管理する
 */
var ReservationService = {
  _APP_FOLDER_NAME: 'WorkflowBuilder AppData',  // メインのアプリデータフォルダ名
  _RESERVATION_FILE_NAME: 'IndividualReservation',
  _RESERVATION_FILE_ID_KEY: 'WORKFLOW_APP_RESERVATION_FILE_ID',

  /**
   * 予約管理シートを取得または新規作成する
   * @private
   * @returns {GoogleAppsScript.Spreadsheet.Spreadsheet}
   */
  _getReservationSheet: function () {
    const userProperties = PropertiesService.getUserProperties();
    let fileId = userProperties.getProperty(this._RESERVATION_FILE_ID_KEY);

    if (fileId) {
      try {
        const file = DriveApp.getFileById(fileId);
        if (!file.isTrashed()) {
          return SpreadsheetApp.openById(fileId);
        }
      } catch (e) {  /* IDが無効な場合は新規作成フローへ */ }
    }

    const appDataFolders = DriveApp.getFoldersByName(this._APP_FOLDER_NAME);
    const appFolder = appDataFolders.hasNext() ? appDataFolders.next() : DriveApp.getRootFolder().createFolder(this._APP_FOLDER_NAME);

    const files = appFolder.getFilesByName(this._RESERVATION_FILE_NAME);
    if (files.hasNext()) {
      const existingFile = files.next();
      userProperties.setProperty(this._RESERVATION_FILE_ID_KEY, existingFile.getId());
      return SpreadsheetApp.openById(existingFile.getId());
    }

    const ss = SpreadsheetApp.create(this._RESERVATION_FILE_NAME);
    const file = DriveApp.getFileById(ss.getId());

    appFolder.addFile(file);
    DriveApp.getRootFolder().removeFile(file);

    userProperties.setProperty(this._RESERVATION_FILE_ID_KEY, ss.getId());

    // 5. ヘッダー行と書式を設定
    const sheet = ss.getSheets()[0];
    sheet.setName('予約リスト');
    const headers = ['ジョブID', 'ステータス', '実行日時', 'ターゲットURL', 'ターゲットシート名', 'ターゲットセル', '設定する値/数式', '実行結果'];
    sheet.appendRow(headers);
    sheet.setFrozenRows(1);

    const headerRange = sheet.getRange('A1:H1');
    headerRange.setBackground('#f3f3f3').setFontWeight('bold').setHorizontalAlignment('center');

    sheet.setColumnWidth(1, 150); // ジョブID
    sheet.setColumnWidth(2, 100); // ステータス
    sheet.setColumnWidth(3, 150); // 実行日時
    sheet.setColumnWidths(4, 3, 200); // URL, シート名, セル
    sheet.setColumnWidth(7, 250); // 設定する値
    sheet.setColumnWidth(8, 250); // 実行結果

    const statusRange = sheet.getRange('B2:B');
    const rules = [
      SpreadsheetApp.newConditionalFormatRule().whenTextEqualTo('完了').setBackground('#d9ead3').setRanges([statusRange]).build(),
      SpreadsheetApp.newConditionalFormatRule().whenTextEqualTo('エラー').setBackground('#f4cccc').setRanges([statusRange]).build(),
      SpreadsheetApp.newConditionalFormatRule().whenTextEqualTo('予約中').setBackground('#fff2cc').setRanges([statusRange]).build()
    ];
    sheet.setConditionalFormatRules(rules);

    return ss;
  },

  /**
   * 予約管理シートのURLを取得する
   * @returns {string} スプレッドシートのURL
   */
  getReservationSheetUrl: function () {
    const ss = this._getReservationSheet();
    return ss.getUrl();
  },

  /**
 * 予約管理シートに新しいジョブを追加する
 * @param {object} jobData - モジュールから渡される設定オブジェクト
 */
  addJob: function (jobData) {
    const sheet = this._getReservationSheet().getSheetByName('予約リスト');
    const jobId = `job-${new Date().getTime()}-${Math.random().toString(36).substring(2, 8)}`;
    const status = '予約中';

    const timeString = jobData.executionTime;
    if (!timeString || typeof timeString !== 'string' || timeString.trim() === '') {
      throw new Error('実行日時が空か、無効な形式です。変数の指定が正しいか確認してください。');
    }

    const executionDateTime = new Date(timeString);

    if (isNaN(executionDateTime.getTime())) {
      throw new Error(`実行日時の形式が無効です: "${timeString}"。日付操作モジュールからの値が正しく渡されているか確認してください。`);
    }

    sheet.appendRow([
      jobId,
      status,
      executionDateTime,
      jobData.targetUrl,
      jobData.targetSheetName,
      jobData.targetCell,
      jobData.valueToSet,
    ]);

    return `予約ジョブ [${jobId}] を追加しました。`;
  }
};
